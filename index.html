#!/usr/bin/env python3 """ MolAkriti — Optical Chemical Structure Recognition (OCSR) Software Outputs Chemsketch-compatible MDL Molfile and SMILES. Uses DECIMER for recognition + RDKit for molecule output + OpenCV for preprocessing. Capture modes ------------- * Open Image – load a file from disk * Crop from File – interactive crop of the currently loaded image * Screenshot – Dynacrop‑style live rubber‑band overlay (DPI‑aware) * Lasso Crop – freehand polygonal selection (click points, double‑click or Enter to finish) All DPI scaling issues resolved – screenshot correctly fills overlay, selected region maps to physical pixels, preview updates reliably. Parameters ---------- image_path : str Path to input image. invert : bool Invert colours (white on black -> black on white). threshold : int Global binarization threshold (0-255) – used only if adaptive=False. adaptive : bool Use adaptive thresholding (Gaussian) instead of global. adaptive_block : int Block size for adaptive thresholding (odd, >1). adaptive_C : int Constant subtracted from mean. denoise_method : str Either 'median' (fast) or 'nlmeans' (slow, strong). Returns ------- str : path to preprocessed temporary PNG file. """ img = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE) if img is None: raise ValueError(f"Cannot read image: {image_path}") h, w = img.shape if w < 800: scale = 800.0 / w img = cv2.resize(img, (int(w * scale), int(h * scale)), interpolation=cv2.INTER_CUBIC) if invert: img = cv2.bitwise_not(img) # Binarization if adaptive: # Ensure block size is odd and at least 3 block = adaptive_block if adaptive_block % 2 == 1 else adaptive_block + 1 if block < 3: block = 3 img = cv2.adaptiveThreshold(img, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, block, adaptive_C) else: _, img = cv2.threshold(img, threshold, 255, cv2.THRESH_BINARY) # Denoising if denoise_method == 'nlmeans': # fastNlMeansDenoising works on grayscale img = cv2.fastNlMeansDenoising(img, None, h=10, templateWindowSize=7, searchWindowSize=21) elif denoise_method == 'median': img = cv2.medianBlur(img, 3) tmp = tempfile.NamedTemporaryFile(suffix='.png', delete=False) cv2.imwrite(tmp.name, img) return tmp.name def recognize_structure(image_path, preprocess_opts=None): if preprocess_opts and preprocess_opts.get('enable', False): proc_path = preprocess_image( image_path, invert=preprocess_opts.get('invert', False), threshold=preprocess_opts.get('threshold', 200), adaptive=preprocess_opts.get('adaptive', False), adaptive_block=preprocess_opts.get('adaptive_block', 11), adaptive_C=preprocess_opts.get('adaptive_C', 2), denoise_method=preprocess_opts.get('denoise_method', 'median'), ) else: proc_path = image_path try: smiles = predict_SMILES(proc_path) if not smiles or not smiles.strip(): raise ValueError("Empty SMILES returned") mol = Chem.MolFromSmiles(smiles) if mol is None: raise ValueError("RDKit could not parse SMILES") return mol, proc_path, smiles except Exception as exc: raise RuntimeError(f"Recognition failed: {exc}") from exc finally: if (preprocess_opts and preprocess_opts.get('enable', False) and proc_path != image_path and os.path.exists(proc_path)): os.unlink(proc_path) def write_molfile(mol, output_path): with open(output_path, 'w') as fh: fh.write(Chem.MolToMolBlock(mol)) def write_smiles(smiles, output_path): with open(output_path, 'w') as fh: fh.write(smiles) # ------------------------------------------------------------------ # Crop dialog (used by "Crop from File" only) # ------------------------------------------------------------------ def _crop_pil_image(self, pil_img, source_name="image"): """Open an interactive crop dialog for an already-loaded PIL image.""" crop_win = tk.Toplevel(self.root) crop_win.title(f"Crop {source_name} — drag to select, double-click to confirm") crop_win.configure(bg='#f0f0f0') sw = crop_win.winfo_screenwidth() sh = crop_win.winfo_screenheight() max_w, max_h = int(sw * 0.9), int(sh * 0.9) iw, ih = pil_img.size scale = 1.0 if iw > max_w or ih > max_h: scale = min(max_w / iw, max_h / ih) display_img = pil_img.resize( (int(iw * scale), int(ih * scale)), Image.Resampling.LANCZOS) else: display_img = pil_img.copy() photo = ImageTk.PhotoImage(display_img) canvas = tk.Canvas(crop_win, width=display_img.width, height=display_img.height, cursor="cross", bg='white') canvas.pack(padx=10, pady=10) canvas.create_image(0, 0, anchor=tk.NW, image=photo) rect_id = None start_x = start_y = None result_path = [None] def on_press(event): nonlocal start_x, start_y, rect_id start_x = canvas.canvasx(event.x) start_y = canvas.canvasy(event.y) if rect_id: canvas.delete(rect_id) rect_id = canvas.create_rectangle( start_x, start_y, start_x, start_y, outline='red', width=2, dash=(4, 2)) def on_drag(event): if start_x is None or rect_id is None: return canvas.coords(rect_id, start_x, start_y, canvas.canvasx(event.x), canvas.canvasy(event.y)) def do_crop(event=None): if rect_id is None: return x1, y1, x2, y2 = canvas.coords(rect_id) if abs(x2 - x1) < 5 or abs(y2 - y1) < 5: messagebox.showwarning( "Crop", "Selection too small. You can now recognise the structure.") # ------------------------------------------------------------------ # Rectangular Screenshot — full DPI‑aware overlay with closures # ------------------------------------------------------------------ def rectangular_screen_capture(self): """Minimise -> grab screen -> dark veil overlay -> rubber-band select -> crop -> restore -> auto-recognise.""" self.root.update_idletasks() self.root.iconify() self.root.after(500, self._start_rect_capture) def _start_rect_capture(self): """Capture primary monitor, build overlay, set up rubber-band selection (DPI‑aware).""" # Capture full screen (primary monitor) – physical pixels full_img = ImageGrab.grab() phys_w, phys_h = full_img.size # Logical screen dimensions (DPI‑scaled) log_w = self.root.winfo_screenwidth() log_h = self.root.winfo_screenheight() # Scaling factors (physical → logical) scale_x = phys_w / log_w scale_y = phys_h / log_h # Resize screenshot to logical size for display display_img = full_img.resize((log_w, log_h), Image.Resampling.LANCZOS) # Overlay window overlay = tk.Toplevel() overlay.overrideredirect(True) overlay.withdraw() overlay.attributes('-alpha', 0.72) overlay.attributes('-topmost', True) overlay.configure(bg='black') overlay.geometry(f"{log_w}x{log_h}+0+0") overlay.deiconify() overlay.focus_force() overlay.update_idletasks() # Canvas canvas = tk.Canvas(overlay, width=log_w, height=log_h, highlightthickness=0, cursor='crosshair', bg='black') canvas.place(x=0, y=0) photo = ImageTk.PhotoImage(display_img) canvas.create_image(0, 0, anchor=tk.NW, image=photo) canvas.image = photo # keep reference # Instruction banner canvas.create_rectangle(0, 0, log_w, 46, fill='#111111', outline='') canvas.create_text(log_w // 2, 23, text="Click and drag to select the chemical structure. " "Release mouse to capture. keep reference # Instruction banner canvas.create_rectangle(0, 0, log_w, 46, fill='#111111', outline='') canvas.create_text(log_w // 2, 23, text="Press and hold LEFT button, DRAG to draw lasso. Release to crop. ESC = cancel", fill='white', font=('Segoe UI', 12, 'bold'), anchor='center') drawing = False points = [] # logical (x,y) on overlay current_line = None def on_press(event): nonlocal drawing, current_line drawing = True points.clear() canvas.delete("lasso") x, y = event.x, event.y points.append((x, y)) current_line = canvas.create_line(x, y, x, y, fill='#FF3333', width=3, tags="lasso") print("[Lasso] Pressed at", x, y) def on_motion(event): nonlocal current_line if not drawing: return x, y = event.x, event.y points.append((x, y)) if current_line: coords = canvas.coords(current_line) new_coords = list(coords) + [x, y] canvas.coords(current_line, *new_coords) def on_release(event): nonlocal drawing, current_line print("[Lasso] Released, points:", len(points)) if not drawing: overlay.destroy() self._restore_window() return drawing = False on_motion(event) # add final point try: if len(points) < 5: print("[Lasso] Too few points, aborting") canvas.create_text(log_w//2, log_h//2, text="Lasso too short. Press ESC to cancel.", fill='yellow', font=('Segoe UI', 16, 'bold'), tags="lasso") overlay.after(2000, overlay.destroy) self._restore_window() return # ---- 1. Bounding box in logical coordinates ---- xs = [p[0] for p in points] ys = [p[1] for p in points] x1_log, x2_log = min(xs), max(xs) y1_log, y2_log = min(ys), max(ys) # ---- 2. Map to physical pixels ---- x1_phys = max(0, int(x1_log * scale_x)) y1_phys = max(0, int(y1_log * scale_y)) x2_phys = min(phys_w, int(x2_log * scale_x)) y2_phys = min(phys_h, int(y2_log * scale_y)) if x2_phys - x1_phys < 10 or y2_phys - y1_phys < 10: print("[Lasso] Bounding box too small") canvas.create_text(log_w//2, log_h//2, text="Selection too small.", fill='yellow', font=('Segoe UI', 16, 'bold')) overlay.after(2000, overlay.destroy) self._restore_window() return # ---- 3. Crop rectangle from full image ---- cropped_rect = full_img.crop((x1_phys, y1_phys, x2_phys, y2_phys)) print(f"[Lasso] Cropped rectangle: {cropped_rect.size}") # ---- 4. Transform lasso points to crop coordinates ---- poly_phys = [(int(p[0] * scale_x), int(p[1] * scale_y)) for p in points] poly_rel = [(px - x1_phys, py - y1_phys) for (px, py) in poly_phys] # Clip to crop bounds poly_rel = [(max(0, min(x, cropped_rect.width-1)), max(0, min(y, cropped_rect.height-1))) for (x, y) in poly_rel] # ---- 5. Create mask (white inside lasso) ---- mask = Image.new('L', cropped_rect.size, 0) ImageDraw.Draw(mask).polygon(poly_rel, outline=1, fill=255) # ---- 6. Paste cropped image onto white background using mask ---- result = Image.new('RGB', cropped_rect.size, (255, 255, 255)) result.paste(cropped_rect, mask=mask) # ---- 7. Save temporary file ---- tmp = tempfile.NamedTemporaryFile(suffix='.png', delete=False) result.save(tmp.name, 'PNG') tmp.close() print("[Lasso] Saved to", tmp.name) # ---- 8. Clean up and proceed ---- overlay.destroy() self._restore_window() self._load_captured_image(tmp.name, label="Lasso crop") self.recognize() except Exception as e: print("[Lasso] Error in on_release:", e) import traceback traceback.print_exc() overlay.destroy() self._restore_window() messagebox.showerror("Lasso Error", str(e)) def on_cancel(event=None): print("[Lasso] Cancelled") overlay.destroy() self._restore_window() canvas.bind('', on_press) canvas.bind('', on_motion) canvas.bind('', on_release) overlay.bind('', on_cancel) #
